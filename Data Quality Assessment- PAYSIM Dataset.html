# Examing correlations between continuous features

# Pearson correlation for continuous variables
corr_matrix = data[['amount', 'oldbalanceOrg', 'newbalanceOrig', 'oldbalanceDest', 'newbalanceDest']].corr()

# Visualize the correlation matrix
sns.heatmap(corr_matrix, annot=True, cmap='coolwarm')
plt.title('Pearson Correlation Heatmap for Continuous Variables')
plt.show()

# There's a strong correlation (0.98) between oldbalanceDest and newbalanceDest.
# This makes sense because these two features represent the balance of the
# destination account before and after the transaction.
# The strong correlation indicates that the two variables are nearly
# redundant we can consider dropping one.
